CREATE TABLE committees (
    committee_id INT AUTO_INCREMENT,
    name VARCHAR(100),
    PRIMARY KEY (committee_id)
);

select
   m.member_id,
   m.name as members,
   c.committee_id,
   c.name as committees
from 
   members m
inner join committees c on c.name=m.name;   

select
   m.member_id,
   m.name as members,
   c.committee_id,
   c.name as committees
from 
   members m
left join committees c on c.name=m.name;   

select
   m.member_id,
   m.name as members,
   c.committee_id,
   c.name as committees
from 
   members m
right join committees c on c.name=m.name;   

select
   m.member_id,
   m.name as members,
   c.committee_id,
   c.name as committees
from 
   members m
cross join committees c;
